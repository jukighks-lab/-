## 형식
    1. 프로젝트 개요
    
         본 프로젝트는 자세가 불안정할때 콘솔로 자세를 정상 주의 비정상으로 표현하기 위해 시작함
         
    2. 프로젝트 구조 & 리소스
    
         노트북이나 웹캠등 카메라로부터의 거리를 인식하기 위해 코딩을 하엿슴
        거리의 공식으로 D = {F * H}/{h} 의 공식으로 거리를 측정하엿슴
        얼굴탐지를 위해여 opencv을 사용함
    
    3. 개발 내용
    
      전 개발일지1에서의 코드가 초록색의 물건이 있을떄에 적용이 되어 그부분을 수정함
      또한 거리간의 인식이 하나의 카메라 일때에는 제대로 측정하기 힘들어 딥러닝기반으로 짜여진 yolo을 이용하여 
      더욱더 정확한 거리측정을 하고자 연결하엿슴 또한 거리의 공식을 변화
      
    4. TMI
    
      먹은건 돈까스 정식
      
> 자료 등이 없어 첨부가 불가능한 메뉴는 기재 X

## 소스 코드
#include <opencv2/opencv.hpp>
#include <iostream>
#include <vector>

using namespace cv;
using namespace cv::dnn;
using namespace std;

int main() {
    // 1. YOLO26 ONNX 모델 불러오기 (파일 이름 변경)
    string model_path = "yolo26n.onnx";
    Net net;
    
    try {
        net = readNetFromONNX(model_path);
        cout << "YOLO26 모델 로드 성공! 코드가 훨씬 간결해졌습니다." << endl;
    } catch (const Exception& e) {
        cerr << "에러: yolo26n.onnx 파일을 찾을 수 없습니다." << endl;
        return -1;
    }

    // 2. 웹캠 연결
    VideoCapture cap(0);
    if (!cap.isOpened()) {
        cerr << "웹캠을 열 수 없습니다." << endl;
        return -1;
    }

    Mat frame;
    cout << "사람 인식 시작 (종료: ESC)" << endl;

    while (true) {
        cap >> frame;
        if (frame.empty()) break;

        // 3. 이미지 전처리 (640x640 크기로 변환)
        Mat blob;
        blobFromImage(frame, blob, 1.0 / 255.0, Size(640, 640), Scalar(), true, false);
        net.setInput(blob);

        // 4. 모델 추론
        vector<Mat> outputs;
        net.forward(outputs, net.getUnconnectedOutLayersNames());

        // 5. 결과 파싱 (NMS-free 모델은 출력 형태가 [1, 300, 6]으로 나옵니다)
        // 300은 최대 객체 수, 6은 [x1, y1, x2, y2, 신뢰도, 클래스ID] 를 의미합니다.
        Mat output = outputs[0];
        int rows = output.size[1]; // 보통 300
        float* data = (float*)output.data;

        // 원본 화면 비율 계산
        float x_scale = (float)frame.cols / 640.0;
        float y_scale = (float)frame.rows / 640.0;

        // 겹치는 박스 제거(NMS) 과정이 통째로 사라졌습니다!
        for (int i = 0; i < rows; ++i) {
            // 데이터 추출 (YOLO26 출력 포맷 기준)
            float x1       = data[i * 6 + 0];
            float y1       = data[i * 6 + 1];
            float x2       = data[i * 6 + 2];
            float y2       = data[i * 6 + 3];
            float score    = data[i * 6 + 4]; // 신뢰도(확률)
            int class_id   = int(data[i * 6 + 5]); // 객체 종류

            // 신뢰도가 50%(0.5) 이상이면서, 객체가 '사람(class_id == 0)'일 때만 그림
            if (score > 0.5 && class_id == 0) {
                
                // 640x640 기준의 좌표를 원본 카메라 화질 크기로 복원
                int left   = int(x1 * x_scale);
                int top    = int(y1 * y_scale);
                int width  = int((x2 - x1) * x_scale);
                int height = int((y2 - y1) * y_scale);

                Rect box(left, top, width, height);

                // 6. 화면에 초록색 사각형과 텍스트 그리기
                rectangle(frame, box, Scalar(0, 255, 0), 2);
                
                string label = "Person: " + to_string(score).substr(0, 4);
                
                // 글씨 배경 박스
                int baseLine;
                Size labelSize = getTextSize(label, FONT_HERSHEY_SIMPLEX, 0.6, 2, &baseLine);
                rectangle(frame, Point(box.x, box.y - labelSize.height - 10), 
                          Point(box.x + labelSize.width, box.y), Scalar(0, 255, 0), FILLED);
                
                putText(frame, label, Point(box.x, box.y - 5), 
                        FONT_HERSHEY_SIMPLEX, 0.6, Scalar(0, 0, 0), 2);
            }
        }

        // 7. 영상 출력
        imshow("YOLO26 Human Detection", frame);

        if (waitKey(1) == 27) break;
    }

    cap.release();
    destroyAllWindows();
    return 0;
}
